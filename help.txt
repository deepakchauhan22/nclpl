 constructor(props) {
    super(props);
    this.state = {
      islogout: false
    };
  }
  signOut = (e) => {
    e.preventDefault();
    localStorage.removeItem("token");
    this.setState({
      islogout: true
      
    });
  };

  if (this.state.islogout) {
      return <Redirect to="/login" />;
    }
